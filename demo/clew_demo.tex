\section*{Clew presentation layer — demo}

\clewExplainer

\noindent
Inline value substitution (SSoT — the number IS the registered value), with
the verified marker rendered green:
\clewval{results_perpatient_auc}.

\medskip\noindent
A claim whose source has drifted renders \textbf{red}, regardless of the
verify gate's verdict:
\clewval{table1_event_completion}.

\medskip\noindent
A verified claim carrying a \emph{frozen} lineage badge:
\clewval{methods_sequential_predictor}.

\medskip\noindent
A verified claim carrying an \emph{exception} badge:
\clewval{discussion_effect_sizes}.

\medskip\noindent
Marking author prose (not a substituted value) with \verb|\clewmark|:
\clewmark{abstract_strongest_patients}{Our strongest patients were P03, P13,
and P11.}

\medskip\noindent
An unknown claim id degrades to a visible warning marker:
\clewval{not_a_real_claim}.

\end{document}
